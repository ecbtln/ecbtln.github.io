126          """ 
127          Due to integer cutoffs, the number of senders and receivers might not be consistent. This take the integer part 
128          of each of the inputs and then assign the remaining few leftovers (so that the sum is the sum of the original 
129          floats) in a way such that the numbers with higher decimal parts will get the extra int before those with lower. 
130   
131          @param unrounded_frequencies: an iterable of floats representing the unrounded frequencies of players playing 
132              each strategy 
133          @type unrounded_frequencies: iterable 
134          @return: an iterable of frequencies of the same length, that sums to the same total. 
135          @rtype: iterable 
136          """ 
137          unrounded_total = math.fsum(unrounded_frequencies) 
138          total = int(round(unrounded_total, DECIMAL_PRECISION)) 
139   
140          int_num_senders = [int(x) for x in unrounded_frequencies] 
141   
142          diff = total - sum(int_num_senders) 
143          if diff > 0: 
144              # note the difference needs to be negative, because heapq's only implement a minimum priority queue but 
145              # we want max priority queue 
146              thresh = [((x - y), i) for i, (x, y) in enumerate(zip(int_num_senders, unrounded_frequencies))] 
147              heapq.heapify(thresh) 
148              while diff > 0: 
149                  v, i = heapq.heappop(thresh) 
150                  int_num_senders[i] += 1 
151                  diff -= 1 
152          assert sum(int_num_senders) == total, "the total number of individuals after rounding must be the same as " \ 
153                                                "before rounding" 
154   
155          return int_num_senders
